# 01.В базе данных Redis подберите коллекцию для подсчета посещений с определенных IP-адресов.

127.0.0.1:6379> HINCRBY ips '127.0.0.1' 1
(integer) 1
127.0.0.1:6379> HGETALL ips
1) "127.0.0.1"
2) "1"
127.0.0.1:6379> HINCRBY ips '127.1.1.1' 1
(integer) 1
127.0.0.1:6379> HGETALL ips
1) "127.0.0.1"
2) "1"
3) "127.1.1.1"
4) "1"

1) "127.0.0.1"
2) "2"
3) "127.1.1.1"
4) "1"

# 02.При помощи базы данных Redis решите задачу поиска имени пользователя по электронному адресу и наоборот, поиск электронного адреса пользователя по его имени.

127.0.0.1:6379> HMSET users 'Ivan' 'ivan@ivan.ru' 'John' 'John@john.com' 'Kate' 'kate@kate.com'
OK
127.0.0.1:6379> HGETALL users
1) "Ivan"
2) "ivan@ivan.ru"
3) "John"
4) "John@john.com"
5) "Kate"
6) "kate@kate.com"
127.0.0.1:6379> HGET users Ivan
"ivan@ivan.ru"

127.0.0.1:6379> HMSET emails 'ivan@ivan.ru' 'Ivan' 'john@john.com' 'John' 'kate@kate.com' 'Kate'
OK
127.0.0.1:6379> HGETALL emails
1) "ivan@ivan.ru"
2) "Ivan"
3) "john@john.com"
4) "John"
5) "kate@kate.com"
6) "Kate"
127.0.0.1:6379> HGET emails kate@kate.com
"Kate"
